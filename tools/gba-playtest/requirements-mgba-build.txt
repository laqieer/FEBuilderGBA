# Pinned, hash-locked Python build inputs for the mGBA 0.10.5 binding.
#
# These are consumed ONLY by the explicit bootstrap scripts under scripts/
# (install-mgba-playtest.ps1 / .sh). The playtest runtime never installs them.
#
# This is STAGE 2 of a two-stage install. Stage 1
# (requirements-mgba-bootstrap.txt) installs the hash-verified pure-Python
# wheels for setuptools/wheel/pytest-runner FIRST. Then these source deps are
# installed with:
#   pip install --require-hashes --no-build-isolation --no-binary :all: \
#       -r requirements-mgba-build.txt
#
# --no-build-isolation reuses the stage-1 (and ensurepip) build backend so no
# un-hashed build backend is fetched from the network.
# --no-binary :all: forces source builds, so only these sdist hashes are needed;
# the C extension is compiled with the validated local toolchain.
#
# Hashes are the official PyPI sdist SHA-256 digests for the pinned versions.

pycparser==2.22 \
    --hash=sha256:491c8be9c040f5390f5bf44a5b07752bd07f56edf992381b05c701439eec10f6

# cffi 2.0 was the first release with Python 3.14 support, but it carries a
# MinGW-specific atomic-store regression that breaks UCRT64 builds. cffi 2.1
# (upstream PR #198) switches that store to GCC/Clang builtin atomics and is
# required here: MSYS2 UCRT64 is a rolling distribution and currently ships
# Python 3.14, so an older 2.0 pin keeps both problems (only 2.1+ fixes the
# MinGW regression while still supporting 3.14).
cffi==2.1.0 \
