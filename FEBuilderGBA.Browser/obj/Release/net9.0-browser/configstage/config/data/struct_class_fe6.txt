// Class struct layout for FE6 (Binding Blade)
// Size: 72 bytes (0x48)
// Field offsets match WinForms ClassForm.Designer.cs control naming convention (B{offset}).
// FE6 has a shorter struct than FE7/8: no rain/snow move costs, promo gains only HP/Str.
@Class_FE6 48
0x00	W	NameTextID	Class name text ID
0x02	W	DescTextID	Class description text ID
0x04	B	ClassNumber	Class number / ID
0x05	B	PromotionClass	Promotion target class
0x06	B	WaitIcon	Map sprite (idle) index
0x07	B	WalkSpeed	Walking speed on map
0x08	W	PortraitID	Generic soldier portrait ID
0x0A	B	SortOrder	Sort order in class list
0x0B	B	BaseHP	Base HP
0x0C	B	BasePow	Base Str/Mag
0x0D	B	BaseSkl	Base Skill
0x0E	B	BaseSpd	Base Speed
0x0F	B	BaseDef	Base Defense
0x10	B	BaseRes	Base Resistance
0x11	B	BaseCon	Base Constitution
0x12	B	BaseMov	Base Movement
0x13	B	MaxHP	Max HP (stat cap)
0x14	B	MaxPow	Max Str/Mag (stat cap)
0x15	B	MaxSkl	Max Skill (stat cap)
0x16	B	MaxSpd	Max Speed (stat cap)
0x17	B	MaxDef	Max Defense (stat cap)
0x18	B	MaxRes	Max Resistance (stat cap)
0x19	B	MaxCon	Max Constitution (stat cap)
0x1A	B	ClassPower	Class power / EXP correction
0x1B	B	GrowthHP	HP growth rate
0x1C	B	GrowthPow	Power growth rate
0x1D	B	GrowthSkl	Skill growth rate
0x1E	B	GrowthSpd	Speed growth rate
0x1F	B	GrowthDef	Defense growth rate
0x20	B	GrowthRes	Resistance growth rate
0x21	B	GrowthLck	Luck growth rate
0x22	B	PromoHP	Promotion HP bonus (CC bonus)
0x23	B	PromoPow	Promotion Str bonus (CC bonus)
0x24	B	Ability1	Ability byte 1
0x25	B	Ability2	Ability byte 2
0x26	B	Ability3	Ability byte 3
0x27	B	Ability4	Ability byte 4
0x28	B	SwordRank	Sword weapon rank
0x29	B	LanceRank	Lance weapon rank
0x2A	B	AxeRank	Axe weapon rank
0x2B	B	BowRank	Bow weapon rank
0x2C	B	StaffRank	Staff weapon rank
0x2D	B	AnimaRank	Anima magic rank
0x2E	B	LightRank	Light magic rank
0x2F	B	DarkRank	Dark magic rank
0x30	P	BattleAnimePointer	Battle animation pointer
0x34	P	MovCostPointer	Movement cost pointer
0x38	P	TerrainAvoidPointer	Terrain avoid bonus pointer
0x3C	P	TerrainDefPointer	Terrain defense bonus pointer
0x40	P	TerrainResPointer	Terrain resistance bonus pointer
0x44	D	Unknown	Unknown data (D68)
